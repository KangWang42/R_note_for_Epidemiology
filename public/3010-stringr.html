stringr 字符串处理完全指南

实用操作
数据处理
字符串处理
stringr 包是 tidyverse 生态中处理字符串的核心工具，提供统一、直观的函数接口，让字符串操作变得简单高效。
Published

January 14, 2026

为什么选择 stringr？

在 R 语言中处理字符串，传统方法使用 Base R 的 paste()、substr()、gsub() 等函数。这些函数功能强大，但存在以下问题：

  • 参数顺序不一致：有时字符串在前，有时模式在前
  • 命名不统一：函数名称难以记忆
  • 缺少向量化支持：部分操作需要手动循环

stringr 包完美解决了这些问题：

特性 说明
统一前缀 所有函数以 str_ 开头，便于自动补全
参数一致 字符串永远是第一个参数
向量化 天然支持向量操作
管道友好 与 %>% 和 |> 完美配合 
# 安装并加载必要的包
if (!require("stringr")) install.packages("stringr")
if (!require("dplyr")) install.packages("dplyr")
if (!require("tibble")) install.packages("tibble")

library(stringr)
library(dplyr)
library(tibble)

基础字符串操作

获取字符串长度：str_length

str_length() 返回字符串中的字符数量，正确处理 UTF-8 编码的中文字符。

# 基本用法
str_length("Hello World")
[1] 11
# 中文字符
str_length("你好世界")
[1] 4
# 向量操作
fruits <- c("apple", "banana", "cherry")
str_length(fruits)
[1] 5 6 6
# 与 nchar() 对比
nchar("你好") # Base R
[1] 2
str_length("你好") # stringr - 结果相同，但处理 NA 更优雅
[1] 2
# NA 处理
str_length(c("hello", NA, "world"))
[1]  5 NA  5

提取子字符串：str_sub

str_sub() 用于提取或替换字符串的一部分，支持负数索引（从末尾计数）。

text <- "Hello World"

# 提取前5个字符
str_sub(text, 1, 5)
[1] "Hello"
# 提取后5个字符（负数索引）
str_sub(text, -5, -1)
[1] "World"
# 从第7个字符到末尾
str_sub(text, 7)
[1] "World"
# 向量操作
words <- c("apple", "banana", "cherry")
str_sub(words, 1, 3)
[1] "app" "ban" "che"
# 替换子字符串
text2 <- "Hello World"
str_sub(text2, 1, 5) <- "Hi"
text2
[1] "Hi World"

字符串拼接：str_c

str_c() 用于拼接字符串，功能类似 paste0()，但更灵活。

# 基本拼接
str_c("Hello", "World")
[1] "HelloWorld"
# 使用分隔符
str_c("Hello", "World", sep = " ")
[1] "Hello World"
# 向量拼接
first_names <- c("张", "李", "王")
last_names <- c("三", "四", "五")
str_c(first_names, last_names)
[1] "张三" "李四" "王五"
# 折叠为单个字符串
str_c(c("a", "b", "c"), collapse = "-")
[1] "a-b-c"
# 处理 NA（默认传播 NA）
str_c("Hello", NA, "World")
[1] NA
# 组合使用 sep 和 collapse
str_c(c("A", "B"), c("1", "2"), sep = "-", collapse = "; ")
[1] "A-1; B-2"

去除空白：str_trim 和 str_squish

# 带空白的字符串
messy <- "   Hello   World   "

# str_trim：去除两端空白
str_trim(messy)
[1] "Hello   World"
# 只去除左侧或右侧
str_trim(messy, side = "left")
[1] "Hello   World   "
str_trim(messy, side = "right")
[1] "   Hello   World"
# str_squish：去除两端空白 + 压缩内部空白
str_squish(messy)
[1] "Hello World"
# 实际应用：清理问卷数据
responses <- c("  Yes ", " No", "  Maybe  ", "Yes  ")
str_trim(responses)
[1] "Yes"   "No"    "Maybe" "Yes"

填充与截断：str_pad 和 str_trunc

# str_pad：填充字符串到指定长度
str_pad("42", width = 5, side = "left", pad = "0")
[1] "00042"
# 不同填充方向
str_pad("hi", 5, "left")
[1] "   hi"
str_pad("hi", 5, "right")
[1] "hi   "
str_pad("hi", 5, "both")
[1] " hi  "
# 批量处理编号
ids <- c("1", "12", "123")
str_pad(ids, width = 4, pad = "0")
[1] "0001" "0012" "0123"
# str_trunc：截断过长字符串
long_text <- "这是一段非常长的文本，需要被截断显示"
str_trunc(long_text, 10)
[1] "这是一段非常长..."
# 指定省略号位置
str_trunc(long_text, 15, side = "left")
[1] "...长的文本，需要被截断显示"
str_trunc(long_text, 15, side = "center")
[1] "这是一段非常...要被截断显示"

大小写转换

text <- "Hello World"

# 转换为大写
str_to_upper(text)
[1] "HELLO WORLD"
# 转换为小写
str_to_lower(text)
[1] "hello world"
# 转换为标题格式（每个单词首字母大写）
str_to_title("hello world from r")
[1] "Hello World From R"
# 转换为句子格式（只有第一个字母大写）
str_to_sentence("hello world from r")
[1] "Hello world from r"
# 实际应用：标准化姓名
names <- c("john DOE", "JANE smith", "bob JONES")
str_to_title(names)
[1] "John Doe"   "Jane Smith" "Bob Jones"

模式检测与计数

检测模式：str_detect

str_detect() 检查字符串是否包含特定模式，返回逻辑值。

fruits <- c("apple", "banana", "apricot", "cherry", "avocado")

# 检测是否包含 "a"
str_detect(fruits, "a")
[1]  TRUE  TRUE  TRUE FALSE  TRUE
# 检测是否以 "a" 开头
str_detect(fruits, "^a")
[1]  TRUE FALSE  TRUE FALSE  TRUE
# 检测是否以 "o" 结尾
str_detect(fruits, "o$")
[1] FALSE FALSE FALSE FALSE  TRUE
# 在 dplyr 中筛选数据
tibble(fruit = fruits) %>%
    filter(str_detect(fruit, "^a"))
# A tibble: 3 × 1
  fruit  
  <chr>  
1 apple  
2 apricot
3 avocado
# 否定检测
str_detect(fruits, "a", negate = TRUE)
[1] FALSE FALSE FALSE  TRUE FALSE

计数匹配：str_count

# 统计字母出现次数
str_count("banana", "a")
[1] 3
# 向量操作
str_count(c("apple", "banana", "cherry"), "a")
[1] 1 3 0
# 统计单词数量（空格分隔）
text <- "The quick brown fox jumps over the lazy dog"
str_count(text, "\\w+")
[1] 9
# 统计数字出现次数
str_count("ID123-456-789", "\\d")
[1] 9

定位匹配：str_locate

text <- "The cat sat on the mat"

# 找到第一个匹配的位置
str_locate(text, "at")
     start end
[1,]     6   7
# 找到所有匹配的位置
str_locate_all(text, "at")
[[1]]
     start end
[1,]     6   7
[2,]    10  11
[3,]    21  22
# 实际应用：找到特定模式的位置
email <- "contact@example.com"
str_locate(email, "@")
     start end
[1,]     8   8

提取与替换

提取匹配内容：str_extract

# 提取第一个匹配
texts <- c("ID: 12345", "Code: ABC123", "Ref: XYZ789")

# 提取数字
str_extract(texts, "\\d+")
[1] "12345" "123"   "789"  
# 提取所有匹配
str_extract_all(texts, "\\d+")
[[1]]
[1] "12345"

[[2]]
[1] "123"

[[3]]
[1] "789"
# 提取字母
str_extract(texts, "[A-Z]+")
[1] "ID" "C"  "R" 
# 医学数据示例：提取剂量
prescriptions <- c(
    "Aspirin 100mg daily",
    "Metformin 500mg twice",
    "Vitamin D 1000IU"
)
str_extract(prescriptions, "\\d+")
[1] "100"  "500"  "1000"

替换内容：str_replace

# 替换第一个匹配
text <- "The cat and the cat"
str_replace(text, "cat", "dog")
[1] "The dog and the cat"
# 替换所有匹配
str_replace_all(text, "cat", "dog")
[1] "The dog and the dog"
# 使用正则表达式替换
dates <- c("2024/01/15", "2024/02/20", "2024/03/25")
str_replace_all(dates, "/", "-")
[1] "2024-01-15" "2024-02-20" "2024-03-25"
# 使用分组进行复杂替换
str_replace(dates, "(\\d{4})/(\\d{2})/(\\d{2})", "\\3-\\2-\\1")
[1] "15-01-2024" "20-02-2024" "25-03-2024"
# 批量替换多个模式
messy_text <- "Hello!!! How are you???"
messy_text %>%
    str_replace_all("!", ".") %>%
    str_replace_all("\\?", ".")
[1] "Hello... How are you..."

删除匹配内容：str_remove

# 删除第一个匹配
str_remove("Hello World World", "World")
[1] "Hello  World"
# 删除所有匹配
str_remove_all("Hello World World", "World")
[1] "Hello  "
# 删除空格
str_remove_all("H e l l o", " ")
[1] "Hello"
# 删除特殊字符
str_remove_all("Price: $100.00", "[^\\d.]")
[1] "100.00"

分割与合并

分割字符串：str_split

# 基本分割
str_split("a-b-c", "-")
[[1]]
[1] "a" "b" "c"
# 返回矩阵格式（指定列数）
str_split_fixed("a-b-c-d", "-", n = 3)
     [,1] [,2] [,3] 
[1,] "a"  "b"  "c-d"
# 分割多个字符串
addresses <- c("北京市|海淀区|中关村", "上海市|浦东区|陆家嘴")
str_split(addresses, "\\|")
[[1]]
[1] "北京市" "海淀区" "中关村"

[[2]]
[1] "上海市" "浦东区" "陆家嘴"
# 限制分割次数
str_split("a-b-c-d", "-", n = 2)
[[1]]
[1] "a"     "b-c-d"
# 转为数据框（配合 tidyr）
tibble(address = addresses) %>%
    tidyr::separate(address, into = c("省市", "区县", "街道"), sep = "\\|")
# A tibble: 2 × 3
  省市   区县   街道  
  <chr>  <chr>  <chr> 
1 北京市 海淀区 中关村
2 上海市 浦东区 陆家嘴

合并字符串：str_flatten

# 将向量合并为单个字符串
words <- c("apple", "banana", "cherry")
str_flatten(words, collapse = ", ")
[1] "apple, banana, cherry"
# 添加最后一个分隔符
str_flatten(words, collapse = ", ", last = " and ")
[1] "apple, banana and cherry"
# 与 str_c(collapse=) 的区别
# str_flatten 专门用于折叠，更加语义化

字符串排序与去重

# 排序
fruits <- c("cherry", "apple", "banana")
str_sort(fruits)
[1] "apple"  "banana" "cherry"
# 降序排序
str_sort(fruits, decreasing = TRUE)
[1] "cherry" "banana" "apple" 
# 按数字排序
files <- c("file2.txt", "file10.txt", "file1.txt")
str_sort(files) # 字母顺序
[1] "file1.txt"  "file10.txt" "file2.txt" 
str_sort(files, numeric = TRUE) # 数字顺序
[1] "file1.txt"  "file2.txt"  "file10.txt"
# 获取排序后的索引
str_order(fruits)
[1] 2 3 1
# 去重
str_unique(c("a", "b", "a", "c", "b"))
[1] "a" "b" "c"

案例一：清理姓名数据

# 模拟杂乱的姓名数据
raw_names <- tibble(
    id = 1:6,
    name = c(
        "  张 三  ",
        "李四",
        "WANG WU",

        "  钱七",

# 清理姓名

    )

案例二：提取医学数据

# 模拟医学记录
medical_records <- tibble(
    patient_id = 1:5,
    record = c(
        "BP: 120/80 mmHg, HR: 72 bpm",
        "Blood Pressure 130/85, Heart Rate 80",
        "收缩压: 125, 舒张压: 82, 心率: 68",
        "SBP=140, DBP=90, Pulse: 88",
        "Blood pressure reading: 115/75 mmHg"
    )
)

# 提取血压和心率数据
extracted <- medical_records %>%
    mutate(
        # 提取收缩压（第一个数字）
        sbp = str_extract(record, "\\d{2,3}(?=/)") %>% as.numeric(),
        # 提取舒张压（斜杠后的数字）
        dbp = str_extract(record, "(?<=/)\\d{2,3}") %>% as.numeric(),
        # 提取心率（各种格式）
        hr = str_extract(record, "(?i)(?<=HR:|heart rate |心率:|Pulse:| )\\d{2,3}") %>%
            as.numeric()
    )

extracted %>% select(patient_id, sbp, dbp, hr)
# A tibble: 5 × 4
  patient_id   sbp   dbp    hr
       <int> <dbl> <dbl> <dbl>
1          1   120    80   120

案例三：标准化地址

# 杂乱地址数据
addresses <- tibble(
    id = 1:4,
    address = c(
        "北京市 海淀区 中关村大街1号",
        "上海市浦东新区陆家嘴环路100号",
        "广州市  天河区  体育西路  200号",
        "深圳市南山区科技园路50号"
    )
)

# 标准化地址
standardized <- addresses %>%
    mutate(
        address_clean = address %>%
            str_squish() %>% # 压缩空白
            str_replace_all("\\s+", ""), # 去除所有空格

        # 提取各级行政区
        province = str_extract(address_clean, "^.+?(?=市)"),
        city = str_extract(address_clean, "(?<=省|^).+?(?=市)"),
        district = str_extract(address_clean, "(?<=市).+?(?=区)"),
        street = str_extract(address_clean, "(?<=区).+")
    )

standardized %>% select(id, province, district, street)

与正则表达式配合

stringr 的所有模式匹配函数都支持正则表达式，详细语法请参考 正则表达式教程。

# 使用 regex() 添加选项
text <- "Hello WORLD"

# 忽略大小写
str_detect(text, regex("world", ignore_case = TRUE))
[1] TRUE
# 多行模式
multiline <- "Line 1\nLine 2\nLine 3"
str_extract_all(multiline, regex("^Line", multiline = TRUE))
[[1]]
[1] "Line" "Line" "Line"
# fixed() 用于精确匹配（不使用正则）
str_detect("a.b.c", fixed(".")) # 匹配字面点号
[1] TRUE
str_detect("a.b.c", ".") # 匹配任意字符
[1] TRUE
