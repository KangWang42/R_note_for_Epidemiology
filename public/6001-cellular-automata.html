get_neighbors <- function(grid, i, j, type = "moore") {
  nrow <- nrow(grid)
  ncol <- ncol(grid)
  
  # å‘¨æœŸæ€§è¾¹ç•Œ
  wrap <- function(x, max) ((x - 1) %% max) + 1
  
  if (type == "moore") {
    # æ‘©å°”é‚»åŸŸ (8é‚»å±…)
    coords <- expand.grid(di = -1:1, dj = -1:1) |> 
      filter(!(di == 0 & dj == 0))
  } else if (type == "von_neumann") {
    # å†¯è¯ºä¾æ›¼é‚»åŸŸ (4é‚»å±…)
    coords <- data.frame(
      di = c(-1, 1, 0, 0),
      dj = c(0, 0, -1, 1)
    )
  }
  
  neighbors <- coords |> 
    mutate(
      ni = wrap(i + di, nrow),
      nj = wrap(j + dj, ncol)
    ) |> 
    rowwise() |> 
    mutate(state = grid[ni, nj]) |> 
    ungroup()
  
  neighbors$state
}

update_grid <- function(grid, rule_func, neighborhood = "moore") {
  nrow <- nrow(grid)
  ncol <- ncol(grid)
  new_grid <- grid
  
  for (i in 1:nrow) {
    for (j in 1:ncol) {
      neighbors <- get_neighbors(grid, i, j, neighborhood)
      new_grid[i, j] <- rule_func(grid[i, j], neighbors)
    }
  }
  
  new_grid
}

}

  }
  
  grid
}

nasch_update <- function(road, vmax = 5, p_slow = 0.3) {
  n <- length(road)
  new_road <- road
  
  for (i in which(road > -1)) {  # åªå¤„ç†æœ‰è½¦çš„å…ƒèƒ
    if (road[i] == -1) next  # -1è¡¨ç¤ºç©ºè½¦ä½
    
    v <- road[i]
    
    # 1. åŠ é€Ÿ
    v <- min(v + 1, vmax)
    
    # 2. å‡é€Ÿ (é¿å…ç¢°æ’)
    gap <- 0
    for (j in 1:vmax) {
      next_pos <- (i + j - 1) %% n + 1
      if (road[next_pos] >= 0) break
      gap <- gap + 1
    }
    v <- min(v, gap)
    
    # 3. éšæœºæ…¢åŒ–
    if (runif(1) < p_slow && v > 0) {
      v <- v - 1
    }
    
    # 4. ç§»åŠ¨
    new_pos <- (i + v - 1) %% n + 1
    new_road[new_pos] <- v
    if (new_pos != i) new_road[i] <- -1
  }
  
  new_road
}
