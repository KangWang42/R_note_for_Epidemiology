ggplot2 样式设置完全指南

可视化教程
ggplot2
样式
进阶
Published

January 14, 2026

前言

本教程专注于 ggplot2 图形样式的精细控制，不讲绘制什么图，而是讲如何把图做得更专业、更美观。涵盖：

  • 🎯 坐标轴：刻度、标签、范围、变换
  • 📊 图例：位置、样式、标题、顺序
  • 📝 标题与文本：标题、副标题、脚注、注释
  • 🎨 美学映射：颜色、大小、形状的精细控制
  • 📐 分面：多面板布局与样式
  • 🖼️ 主题：全面的主题定制
library(ggplot2)
library(dplyr)
library(scales)

# 创建示例数据
set.seed(42)
demo_data <- data.frame(
    group = rep(c("A组", "B组", "C组", "D组"), each = 50),
    x = c(rnorm(50, 10, 2), rnorm(50, 15, 3), rnorm(50, 20, 2.5), rnorm(50, 25, 4)),
    y = c(rnorm(50, 100, 15), rnorm(50, 150, 20), rnorm(50, 200, 25), rnorm(50, 250, 30)),
    size = runif(200, 1, 10),
    category = sample(c("类型1", "类型2", "类型3"), 200, replace = TRUE)
)

# 基础图形（后续会不断完善）
p_base <- ggplot(demo_data, aes(x = x, y = y, color = group)) +
    geom_point(size = 3, alpha = 0.7)
p_base

第一部分：坐标轴详细设置

坐标轴是图形最重要的组成部分之一，包括刻度线、刻度标签、轴标签、轴线等。

坐标轴范围与限制

# 方法1：xlim/ylim（会删除范围外的数据）
p_base +
    xlim(5, 30) +
    ylim(50, 300)
# 方法2：coord_cartesian（只是"放大"，不删除数据）
# 推荐用于放大局部区域
p_base +
    coord_cartesian(xlim = c(10, 25), ylim = c(100, 250))
# 方法3：scale_*_continuous 设置（最精细的控制）
p_base +
    scale_x_continuous(limits = c(5, 35), expand = c(0, 0)) +
    scale_y_continuous(limits = c(0, 350), expand = c(0, 0))

坐标轴刻度控制

# 设置主刻度和次刻度
p_base +
    scale_x_continuous(
        breaks = seq(5, 35, by = 5), # 主刻度位置
        minor_breaks = seq(5, 35, by = 1), # 次刻度位置
        limits = c(5, 35)
    ) +
    scale_y_continuous(
        breaks = seq(0, 350, by = 50),
        minor_breaks = seq(0, 350, by = 25),
        limits = c(0, 350)
    )
# 自定义刻度标签
p_base +
    scale_x_continuous(
        breaks = c(10, 15, 20, 25, 30),
        labels = c("低", "中低", "中", "中高", "高") # 自定义标签
    ) +
    scale_y_continuous(
        breaks = seq(100, 300, by = 50),
        labels = function(x) paste0(x, " 单位") # 使用函数
    )
# 使用 scales 包格式化
p_base +
    scale_x_continuous(labels = scales::label_number(suffix = " L")) +
    scale_y_continuous(labels = scales::label_comma()) # 千分位逗号

坐标轴标签样式

p_base +
    labs(x = "X 变量 (单位)", y = "Y 变量 (单位)") +
    theme(
        # X 轴标签
        axis.title.x = element_text(
            size = 14, # 字号
            face = "bold", # 粗体
            color = "#1f2937", # 颜色
            margin = margin(t = 10) # 上边距
        ),
        # Y 轴标签
        axis.title.y = element_text(
            size = 14,
            face = "bold",
            color = "#1f2937",
            margin = margin(r = 10),
            angle = 90 # 旋转角度（默认90）
        )
    )

刻度标签样式

p_base +
    theme(
        # X 轴刻度标签
        axis.text.x = element_text(
            size = 11,
            color = "#374151",
            angle = 45, # 旋转角度
            hjust = 1, # 水平对齐
            vjust = 1 # 垂直对齐
        ),
        # Y 轴刻度标签
        axis.text.y = element_text(
            size = 11,
            color = "#374151"
        )
    )

坐标轴线与刻度线

p_base +
    theme(
        # 坐标轴线
        axis.line = element_line(color = "#1f2937", linewidth = 0.8),
        # 刻度线
        axis.ticks = element_line(color = "#1f2937", linewidth = 0.5),
        # 刻度线长度
        axis.ticks.length = unit(0.25, "cm"),
        # 去掉面板边框（配合轴线使用）
        panel.border = element_blank()
    )

对数坐标轴

# 创建适合对数变换的数据
log_data <- data.frame(
    x = c(1, 10, 100, 1000, 10000),
    y = c(2, 20, 200, 2000, 20000)
)

ggplot(log_data, aes(x, y)) +
    geom_point(size = 4, color = "#4f46e5") +
    geom_line(color = "#4f46e5") +
    scale_x_log10(
        breaks = scales::trans_breaks("log10", function(x) 10^x),
        labels = scales::trans_format("log10", scales::math_format(10^.x))
    ) +
    scale_y_log10(
        breaks = scales::trans_breaks("log10", function(x) 10^x),
        labels = scales::trans_format("log10", scales::math_format(10^.x))
    ) +
    annotation_logticks(sides = "bl") + # 在左下添加对数刻度
    theme_bw()

翻转与极坐标

# 翻转坐标轴（横向柱状图）
ggplot(demo_data, aes(x = group, y = y)) +
    geom_boxplot(fill = "#4f46e5", alpha = 0.7) +
    coord_flip() +
    theme_minimal()
# 极坐标（饼图/雷达图基础）
pie_data <- data.frame(
    category = c("A", "B", "C", "D"),
    value = c(25, 35, 20, 20)
)

ggplot(pie_data, aes(x = "", y = value, fill = category)) +
    geom_bar(stat = "identity", width = 1) +
    coord_polar("y") +
    theme_void() +
    scale_fill_brewer(palette = "Set2")

第二部分：图例详细设置

图例是解读图形的关键，需要精心设计。

图例位置

# 预设位置
p_base + theme(legend.position = "top")
p_base + theme(legend.position = "bottom")
p_base + theme(legend.position = "left")
# 精确位置（坐标 0-1）
p_base +
    theme(
        legend.position = c(0.85, 0.2), # 右下角内部
        legend.background = element_rect(fill = "white", color = "gray80")
    )
# 隐藏图例
p_base + theme(legend.position = "none")

图例标题与标签

p_base +
    labs(color = "实验分组") + # 修改图例标题
    scale_color_discrete(
        labels = c("对照组", "低剂量", "中剂量", "高剂量") # 修改标签
    )
# 去掉图例标题
p_base +
    theme(legend.title = element_blank())

图例样式定制

p_base +
    labs(color = "分组") +
    theme(
        # 图例标题
        legend.title = element_text(
            size = 12, face = "bold", color = "#1f2937"
        ),
        # 图例标签
        legend.text = element_text(
            size = 10, color = "#374151"
        ),
        # 图例背景
        legend.background = element_rect(
            fill = "#f9fafb",
            color = "#e5e7eb",
            linewidth = 0.5
        ),
        # 图例键（图标）背景
        legend.key = element_rect(fill = "white"),
        # 图例键大小
        legend.key.size = unit(1, "cm"),
        # 图例内边距
        legend.margin = margin(5, 10, 5, 10)
    )

图例布局

# 水平排列
p_base +
    theme(legend.position = "bottom") +
    guides(color = guide_legend(nrow = 1)) # 单行
# 自定义列数
p_base +
    theme(legend.position = "bottom") +
    guides(color = guide_legend(ncol = 2)) # 两列

图例顺序与反转

# 反转图例顺序
p_base +
    guides(color = guide_legend(reverse = TRUE))
# 自定义顺序
p_base +
    scale_color_discrete(limits = c("D组", "C组", "B组", "A组"))

多个图例的控制

# 创建有多个映射的图
p_multi <- ggplot(demo_data, aes(x = x, y = y, color = group, size = size, shape = category)) +
    geom_point(alpha = 0.7)

# 单独控制每个图例
p_multi +
    guides(
        color = guide_legend(order = 1, title = "分组"),
        size = guide_legend(order = 2, title = "大小"),
        shape = guide_legend(order = 3, title = "类型")
    ) +
    theme(legend.box = "vertical") # 图例垂直排列

第三部分：标题与注释

完整标题系统

p_base +
    labs(

标题样式定制

添加数学公式

p_base +
    annotate(

添加矩形/线条注释

p_base +
    # 矩形高亮区域
    annotate(
        "rect",
        xmin = 18, xmax = 28,
        ymin = 180, ymax = 320,
        alpha = 0.2,
        fill = "#4f46e5"
    ) +
    # 垂直参考线
    geom_vline(xintercept = 20, linetype = "dashed", color = "#ef4444", linewidth = 1) +
    # 水平参考线
    geom_hline(yintercept = 200, linetype = "dotted", color = "#10b981", linewidth = 1) +
    # 箭头注释
    annotate(
        "segment",
        x = 28, xend = 25,
        y = 100, yend = 180,
        arrow = arrow(length = unit(0.3, "cm")),
        color = "#4f46e5"
    ) +
    annotate("text", x = 28, y = 90, label = "注意这个区域", size = 4)

带标签的数据点

# 只标注特定点
highlight_data <- demo_data %>%
    filter(y > 280 | y < 70)

p_base +
    geom_text(
        data = highlight_data,
        aes(label = round(y, 0)),
        vjust = -1,
        size = 3
    )
# 使用 ggrepel 避免标签重叠
library(ggrepel)

p_base +
    geom_text_repel(
        data = highlight_data,
        aes(label = paste0("y=", round(y, 0))),
        size = 3,
        max.overlaps = 20
    )

# 定义自己的主题函数
theme_my_style <- function(base_size = 12) {
    theme_minimal(base_size = base_size) +
        theme(
            # 标题
            plot.title = element_text(face = "bold", hjust = 0.5),
            plot.subtitle = element_text(color = "gray50", hjust = 0.5),
            # 面板
            panel.grid.minor = element_blank(),
            panel.grid.major = element_line(color = "#f0f0f0"),
            # 坐标轴
            axis.title = element_text(face = "bold"),
            axis.text = element_text(color = "#333333"),
            # 图例
            legend.position = "bottom",
            legend.background = element_rect(fill = "white", color = "gray90"),
            # 分面
            strip.background = element_rect(fill = "#4f46e5"),
            strip.text = element_text(color = "white", face = "bold")
        )
}

小结

本教程涵盖了 ggplot2 样式设置的所有重要方面：

  1. 坐标轴：范围、刻度、标签、轴线、变换
  2. 图例：位置、标题、标签、布局、多图例控制
  3. 标题注释：标题系统、文本注释、几何注释
  4. 美学映射：颜色、大小、形状、透明度、线型
  5. 分面：wrap/grid、标签样式、间距控制
  6. 主题系统：预设主题、完整定制、可复用主题

建议：保存常用的 theme() 设置为函数，在所有图形中复用，保持风格统一。
